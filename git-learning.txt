#For stashing one file:
	git stash -- filename.txt
#For stashing more than one file:
	git stash -m "stash-message" -- filename1.txt filename2.txt
#if I want to only stash changes in the src/ directory
git stash push src/

git stash save "message"         // create stash,
git stash push -m "message" // create stash with msg,

git stash pop // apply and remove stash entry
git stash apply         // to apply stash,
git stash apply indexno // to apply  specific stash, 
git stash list          //list stash,
git stash drop indexno      //to delete stash,
git stash pop indexno,
git stash pop = stash apply + stash drop
git stash clear         //clear stash entries and delete all your local stashed code
git stash show      // show the top stashed changes as below
$ git stash show
